apiVersion: argoproj.io/v1alpha1
kind: Application
metadata:
  name: argocd-mockup
  namespace: n8n
spec:
  project: n8n
  source:
    repoURL: "https://github.com/myrepo/app.git"
    targetRevision: main
    path: dev
    directory:
      recurse: true
      jsonnet:
        libs:
          - lib
  destination:
    server: https://kubernetes.default.svc
    namespace: default
  syncPolicy:
    automated:
      prune: false
      selfHeal: true
